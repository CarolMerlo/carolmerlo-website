# Objective

A dismissible popup modal on the homepage offering a free meditation recording in exchange for an email signup. Powered by Mailchimp. Appears 10 seconds after page load, shown once per visitor (30-day cookie). Footer also gets a "Subscribe" link that reopens the popup.

## 4. JS — add to `main.js` inside `DOMContentLoaded`:
```javascript
/* Email popup */
(function () {
  var COOKIE = 'cm_popup_seen';
  var DELAY  = 10000; // 10 seconds

  function getCookie(name) {
    return document.cookie.split('; ').some(function (c) {
      return c.startsWith(name + '=');
    });
  }
  function setCookie(name, days) {
    var exp = new Date(Date.now() + days * 864e5).toUTCString();
    document.cookie = name + '=1; expires=' + exp + '; path=/';
  }

  var popup   = document.getElementById('subscribe-popup');
  var trigger = document.getElementById('footer-subscribe-trigger');

  if (!popup) return;

  function openPopup() {
    popup.removeAttribute('hidden');
    document.body.style.overflow = 'hidden';
  }
  function closePopup() {
    popup.setAttribute('hidden', '');
    document.body.style.overflow = '';
    setCookie(COOKIE, 30);
  }

  // Close button
  popup.querySelector('.popup-close').addEventListener('click', closePopup);
  // Click outside modal
  popup.addEventListener('click', function (e) {
    if (e.target === popup) closePopup();
  });
  // Escape key
  document.addEventListener('keydown', function (e) {
    if (e.key === 'Escape' && !popup.hasAttribute('hidden')) closePopup();
  });

  // Footer trigger (all pages — only fires popup on homepage)
  if (trigger) {
    trigger.addEventListener('click', function (e) {
      e.preventDefault();
      openPopup();
    });
  }

  // Auto-open after delay (homepage only, once per 30 days)
  if (!getCookie(COOKIE)) {
    setTimeout(openPopup, DELAY);
  }
}());
```
